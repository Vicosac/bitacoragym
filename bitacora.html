function toast(msg, ok=true) {

  const el = document.getElementById('toast');

  el.textContent = msg;

  el.style.background = ok ? 'var(--accent)' : 'var(--red)';

  el.style.color = ok ? '#000' : '#fff';

  el.classList.add('show');

  setTimeout(() => el.classList.remove('show'), 2500);

}

async function apiFetch(path, options={}) {

  const res = await fetch(`${SUPABASE_URL}/rest/v1/${path}`, { headers: HEADERS, ...options });

  if (!res.ok) { const err = await res.text(); throw new Error(err); }

  const text = await res.text();

  return text ? JSON.parse(text) : [];

}

function addSerie(ejId) {

  const list = document.getElementById(`series-${ejId}`);

  const num = list.querySelectorAll('.serie-row').length + 1;

  const sid = serieCount++;

  const row = document.createElement('div');

  row.className = 'serie-row';

  row.id = `serie-${sid}`;

  row.innerHTML = `

    <span class="serie-num">${num}</span>

    <input type="text" placeholder="ej: 12" class="serie-reps">

    <input type="number" placeholder="kg" class="serie-peso" step="0.5" min="0">

    <button class="btn-remove" onclick="removeSerie('serie-${sid}', ${ejId})">×</button>

  `;

  list.appendChild(row);

}

function removeSerie(sid, ejId) {

  document.getElementById(sid).remove();

  document.getElementById(`series-${ejId}`).querySelectorAll('.serie-row').forEach((r, i) => {

    r.querySelector('.serie-num').textContent = i + 1;

  });

}

function getEjercicios() {

  return [...document.querySelectorAll('.ejercicio-block')].map(block => {

    const nombre = block.querySelector('.ej-nombre').value.trim();

    const series = [...block.querySelectorAll('.serie-row')].map((r, i) => ({

      serie: i + 1,

      reps: r.querySelector('.serie-reps').value.trim(),

      peso: r.querySelector('.serie-peso').value,

    })).filter(s => s.reps || s.peso);

    return { nombre, series };

  }).filter(e => e.nombre);

}

async function guardarSesion() {

  const fecha = document.getElementById('fecha').value;

  if (!fecha) { toast('Ingresá la fecha.', false); return; }

  const btn = document.getElementById('btn-guardar');

  btn.disabled = true; btn.textContent = 'Guardando...';

  try {

    await apiFetch('sesiones', {

      method: 'POST',

      body: JSON.stringify({

        fecha,

        tipo: document.getElementById('tipo').value,

        calor: document.getElementById('calor').value.trim(),

        energia: parseInt(document.getElementById('energia').value),

        ejercicios: getEjercicios(),

        aerobico: getAerobico(),

        sensacion: parseInt(document.getElementById('sensacion').value),

        notas: document.getElementById('notas').value.trim(),

      })

    });

    toast('Sesión guardada');

    document.getElementById('fecha').valueAsDate = new Date();

    document.getElementById('calor').value = '';

    document.getElementById('energia').value = 7; document.getElementById('energia-val').textContent = '7';

    document.getElementById('sensacion').value = 7; document.getElementById('sensacion-val').textContent = '7';

    document.getElementById('notas').value = '';

    document.getElementById('ejercicios-list').innerHTML = '';

    document.getElementById('aerobico-list').innerHTML = '';

    ejCount = 0; serieCount = 0; aeroCount = 0;

  } catch(e) {

    toast('Error al guardar', false); console.error(e);

  }

  btn.disabled = false; btn.textContent = 'Guardar sesión';

}

function showPage(name, btn) {

  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));

  document.querySelectorAll('nav button').forEach(b => b.classList.remove('active'));

  document.getElementById(`page-${name}`).classList.add('active');

  btn.classList.add('active');

  if (name === 'historial') renderHistorial();

  if (name === 'analisis') renderAnalisis();

}

function formatFecha(str) {

  const [y,m,d] = str.split('-'); return `${d}/${m}/${y}`;

}

        </div>

                  ${(e.series||[]).map(sr => `

                    <div class="ej-serie-item">S${sr.serie} — <span>${sr.reps} reps</span>${sr.peso ? ` · <span>${sr.peso}kg</span>` : ''}</div>

                  `).join('')}

                </div>

              </div>

            `).join('')}

          </div>` : ''}

        ${(s.aerobico||[]).length ? `

          <div class="card-subsection">

            <div class="card-subsection-title">Aeróbico</div>

              <div class="aerobico-item">

                ${a.tiempo ? ` · <span>${a.tiempo} min</span>` : ''}

                ${a.calorias ? ` · <span>${a.calorias} kcal</span>` : ''}

            `).join('')}

          </div>` : ''}

async function deleteSesion(id) {

  if (!confirm('¿Eliminar esta sesión?')) return;

  try {

    await apiFetch(`sesiones?id=eq.${id}`, { method: 'DELETE' });

    toast('Sesión eliminada');

    renderHistorial();

  } catch(e) { toast('Error al eliminar', false); }

}

async function renderAnalisis() {

  document.getElementById('stats-grid').innerHTML = '<div class="loading">Cargando...</div>';

  try {

    const sessions = await apiFetch('sesiones?order=fecha.desc');

    const total = sessions.length;

    const avgE = total ? (sessions.reduce((a,s)=>a+s.energia,0)/total).toFixed(1) : '—';

    const avgS = total ? (sessions.reduce((a,s)=>a+s.sensacion,0)/total).toFixed(1) : '—';

    const totalCal = sessions.reduce((a,s) => a + (s.aerobico||[]).reduce((b,ae) => b + (parseInt(ae.calorias)||0), 0), 0);

    document.getElementById('stats-grid').innerHTML = `

      <div class="stat-box"><div class="metric-label">Sesiones totales</div><div class="metric-val">${total}</div></div>

      <div class="stat-box"><div class="metric-label">Energía promedio</div><div class="metric-val">${avgE}</div></div>

      <div class="stat-box"><div class="metric-label">Sensación promedio</div><div class="metric-val">${avgS}</div></div>

      <div class="stat-box"><div class="metric-label">Calorías aeróbico (total)</div><div class="metric-val">${totalCal > 0 ? totalCal.toLocaleString() : '—'}</div></div>

    `;

    const freqEj = {};

    sessions.forEach(s => (s.ejercicios||[]).forEach(e => {

      const k = e.nombre.toLowerCase();

      freqEj[k] = (freqEj[k]||0) + 1;

    }));

    const sortedEj = Object.entries(freqEj).sort((a,b)=>b[1]-a[1]).slice(0,8);

    const maxEj = sortedEj[0]?.[1] || 1;

    document.getElementById('freq-list').innerHTML = sortedEj.length

      ? sortedEj.map(([n,c]) => `

        <div class="freq-item">

          <span style="min-width:160px;text-transform:capitalize">${n}</span>

          <div class="freq-bar-wrap"><div class="freq-bar" style="width:${c/maxEj*100}%"></div></div>

          <span class="freq-count">${c}</span>

        </div>`).join('')

      : '<div style="color:var(--text-muted);font-size:0.82rem">Sin datos aún.</div>';

    const freqAe = {};

    sessions.forEach(s => (s.aerobico||[]).forEach(a => {

      const k = a.tipo.toLowerCase();

      freqAe[k] = (freqAe[k]||0) + 1;

    }));

    const sortedAe = Object.entries(freqAe).sort((a,b)=>b[1]-a[1]).slice(0,6);

    const maxAe = sortedAe[0]?.[1] || 1;

    document.getElementById('freq-aerobico').innerHTML = sortedAe.length

      ? sortedAe.map(([n,c]) => `

        <div class="freq-item">

          <span style="min-width:160px;text-transform:capitalize">${n}</span>

          <div class="freq-bar-wrap"><div class="freq-bar" style="width:${c/maxAe*100}%"></div></div>

          <span class="freq-count">${c}</span>

        </div>`).join('')

      : '<div style="color:var(--text-muted);font-size:0.82rem">Sin datos aún.</div>';

  } catch(e) {

    document.getElementById('stats-grid').innerHTML = '<div class="empty-state">Error al cargar.</div>';

  }

}
